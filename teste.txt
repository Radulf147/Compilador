int main() {
    int x;
    x = 2;
    switch (x) {
        case 1:
            printf("um\n");
            break;
        case 2:
            printf("dois\n");
            break;
        case 3:
            printf("tres\n");
            break;
        default:
            printf("outro\n");
    }
    return 0;
}
